      $(function() {
        var $p = $('.truncated-text');
        $p.truncateMiddle({length: 25});
        setInterval(function() {
          var newLength = Math.floor(Math.random() * 20) + 15
          $p.truncateMiddle('setOptions', {length: newLength});
        }, 1000);
      });
